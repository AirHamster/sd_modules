110 static inline void filterDirectionValues(
111  const float* income,
112  float * filtered,
113  const uint8_t lengthBuf,
114  const uint8_t windwoSize)
115 {
116  static float sineValues[FILTER_BUFFER_SIZE] = {0.0};
117  static float cosineValues[FILTER_BUFFER_SIZE] = {0.0};
118  if(lengthBuf > FILTER_BUFFER_SIZE)
119  {
120  return;
121  }
122  for(uint8_t i = 0; i < lengthBuf; i++)
123  {
124  filtered[i] = sin(radiansFromDegrees(income[i]));
125  }
126  trimmedMeanFilter(filtered, &sineValues[0], lengthBuf, windwoSize, 0.0001);
127  for(uint8_t i = 0; i < lengthBuf; i++)
128  {
129  filtered[i] = cos(radiansFromDegrees(income[i]));
130  }
131  trimmedMeanFilter(filtered, &cosineValues[0], lengthBuf, windwoSize, 0.0001);
132  for(uint8_t i = 0; i < lengthBuf; i++)
133  {
134  filtered[i] = direction(degreesFromRadians(atan2(sineValues[i], cosineValues[i])));
135  }
136 };
